\begin{enumerate}
\item CSIR-UGC-NET(mathA june 2015) Q.110 :\\
Suppose X has density
\begin{align}
f(x|\theta)= \frac{1}{\theta}e^{\frac{-x}{\theta}} , x>0,  \theta >0
\end{align}
Define Y as follows\\
\begin{align}
Y = K if k \leqslant X < k+1, k= 0,1,2,... 
\end{align}
Then the distribution of Y is\\
1.normal\\
2.binomial\\
3.poisson\\
4.geometric
\item Solution\\
given thatx has density f then
\begin{align}
PDF(x)=\frac{e^{-x/\theta}}{{\theta}}
\end{align}
therefore Q will be 
\begin{align}
Q_X(\alpha)=\int_{\alpha}^{\infty} \frac{e^{-x/\theta}}{{\theta}}\\
Q_X(\alpha)=e^{-\alpha/\theta}
\end{align}
Now
\begin{align}
Y = k, if k \leqslant X < k+1
\end{align}
so considering Q
\begin{align}
\Pr(Y=K) = \Pr(k \leqslant X < k+1)\\
=Q_{X}(k) - Q_{X}(k+1)\\
=e^{-k/\theta}-e^{-(K+1)/\theta}\\
=a*e^{-k} \text{where a is constant}
\end{align}
geometric distribution is given by 
\begin{align}
\Pr(X=K)= (1-p)^{k-1}p
\end{align}
therefore obtained y distribution is geometric\\ 
So the correct option is \brak{4}
\end{enumerate}
